```prisma
model Sponsor {
  id          String   @id @default(cuid())
  userId      String

  // Brand info
  name        String
  company     String?
  email       String?
  phone       String?
  website     String?
  logo        String?  // URL to logo

  // Relationship
  status      String   @default("lead") // lead, active, past
  notes       String?

  // Metadata
  metadata    Json?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  // Relations
  user        User           @relation(fields: [userId], references: [id], onDelete: Cascade)
  deals       SponsorDeal[]

  @@index([userId])
  @@index([userId, status])
  @@map("sponsors")
}

model SponsorDeal {
  id          String   @id @default(cuid())
  sponsorId   String

  // Deal info
  title       String
  description String?
  value       Decimal  @db.Decimal(12, 2)
  currency    String   @default("USD")

  // Pipeline
  stage       String   @default("negotiation") // negotiation, contracted, in_progress, delivered, paid, cancelled

  // Dates
  startDate   DateTime?
  endDate     DateTime?
  paidAt      DateTime?

  // Payment tracking
  revenueEntryId String? // Link to RevenueEntry when paid

  // Metadata
  metadata    Json?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  // Relations
  sponsor      Sponsor          @relation(fields: [sponsorId], references: [id], onDelete: Cascade)
  deliverables DealDeliverable[]
  revenueEntry RevenueEntry?    @relation(fields: [revenueEntryId], references: [id])

  @@index([sponsorId])
  @@index([sponsorId, stage])
  @@map("sponsor_deals")
}

model DealDeliverable {
  id          String   @id @default(cuid())
  dealId      String

  // Deliverable info
  title       String
  description String?
  type        String   // post, story, reel, video, mention, review, other
  platform    String?  // YouTube, Instagram, TikTok, etc.

  // Status
  status      String   @default("pending") // pending, in_progress, submitted, approved, rejected
  dueDate     DateTime?
  completedAt DateTime?

  // Content link
  contentUrl  String?  // URL to published content
  postId      String?  // Link to internal Post if applicable

  // Metadata
  metadata    Json?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  // Relations
  deal        SponsorDeal @relation(fields: [dealId], references: [id], onDelete: Cascade)

  @@index([dealId])
  @@index([dealId, status])
  @@map("deal_deliverables")
}
```

Add relations to User model:
```prisma
sponsors             Sponsor[]
```

Add relation to RevenueEntry (optional back-reference):
```prisma
sponsorDeal          SponsorDeal?
```

```typescript
// Types
export type SponsorStatus = 'lead' | 'active' | 'past';
export type DealStage = 'negotiation' | 'contracted' | 'in_progress' | 'delivered' | 'paid' | 'cancelled';
export type DeliverableType = 'post' | 'story' | 'reel' | 'video' | 'mention' | 'review' | 'other';
export type DeliverableStatus = 'pending' | 'in_progress' | 'submitted' | 'approved' | 'rejected';

export const SPONSOR_STATUSES: SponsorStatus[] = ['lead', 'active', 'past'];
export const DEAL_STAGES: DealStage[] = ['negotiation', 'contracted', 'in_progress', 'delivered', 'paid', 'cancelled'];
export const DELIVERABLE_TYPES: DeliverableType[] = ['post', 'story', 'reel', 'video', 'mention', 'review', 'other'];
export const DELIVERABLE_STATUSES: DeliverableStatus[] = ['pending', 'in_progress', 'submitted', 'approved', 'rejected'];

// Interfaces for Sponsor, SponsorDeal, DealDeliverable
// Input interfaces for Create/Update operations

// Service class
export class SponsorService {
  // Sponsors
  async getSponsors(userId: string, filters?): Promise<Sponsor[]>;
  async getSponsor(id: string, userId: string): Promise<Sponsor | null>;
  async createSponsor(userId: string, data): Promise<Sponsor>;
  async updateSponsor(id: string, userId: string, data): Promise<Sponsor>;
  async deleteSponsor(id: string, userId: string): Promise<void>;

  // Deals
  async getDeals(userId: string, filters?): Promise<SponsorDeal[]>;
  async getDeal(id: string, userId: string): Promise<SponsorDeal | null>;
  async createDeal(userId: string, sponsorId: string, data): Promise<SponsorDeal>;
  async updateDeal(id: string, userId: string, data): Promise<SponsorDeal>;
  async deleteDeal(id: string, userId: string): Promise<void>;

  // Deliverables
  async getDeliverables(dealId: string, userId: string): Promise<DealDeliverable[]>;
  async createDeliverable(dealId: string, userId: string, data): Promise<DealDeliverable>;
  async updateDeliverable(id: string, userId: string, data): Promise<DealDeliverable>;
  async deleteDeliverable(id: string, userId: string): Promise<void>;

  // Summary/Pipeline
  async getPipelineSummary(userId: string): Promise<PipelineSummary>;
}
```
