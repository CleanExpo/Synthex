Add three new models for social listening:

1. **TrackedKeyword** - Keywords/hashtags to monitor
```prisma
model TrackedKeyword {
  id              String    @id @default(cuid())
  userId          String
  organizationId  String?

  keyword         String    // The keyword or hashtag (without #)
  type            String    // 'keyword', 'hashtag', 'mention', 'brand'
  platforms       String[]  // Which platforms to monitor
  isActive        Boolean   @default(true)

  // Tracking stats
  lastCheckedAt   DateTime?
  totalMentions   Int       @default(0)

  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt

  user            User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  organization    Organization? @relation(fields: [organizationId], references: [id])
  mentions        SocialMention[]

  @@index([userId])
  @@index([organizationId])
  @@index([keyword])
}
```

2. **SocialMention** - Individual mention/post found
```prisma
model SocialMention {
  id              String    @id @default(cuid())
  keywordId       String
  userId          String

  // Platform info
  platform        String    // 'twitter', 'instagram', etc.
  platformPostId  String    // External post ID
  platformUrl     String?   // Link to original post

  // Author info
  authorHandle    String
  authorName      String?
  authorAvatar    String?
  authorFollowers Int?

  // Content
  content         String    @db.Text
  mediaUrls       String[]

  // Metrics
  likes           Int       @default(0)
  comments        Int       @default(0)
  shares          Int       @default(0)
  reach           Int?      // Estimated reach

  // Analysis
  sentiment       String?   // 'positive', 'negative', 'neutral'
  sentimentScore  Float?    // -1.0 to 1.0
  isInfluencer    Boolean   @default(false)

  // Status
  isRead          Boolean   @default(false)
  isArchived      Boolean   @default(false)
  isFlagged       Boolean   @default(false)

  postedAt        DateTime
  fetchedAt       DateTime  @default(now())
  createdAt       DateTime  @default(now())

  keyword         TrackedKeyword @relation(fields: [keywordId], references: [id], onDelete: Cascade)
  user            User      @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([platform, platformPostId])
  @@index([keywordId])
  @@index([userId])
  @@index([platform])
  @@index([sentiment])
  @@index([postedAt])
}
```

