  canonicalUrl    String

  logoUrl         String?

  address         Json?                    // { street, suburb, state, postcode, country }
  phone           String?
  // sameAs profile URLs (prioritised)
  wikidataUrl     String?
  wikipediaUrl    String?
  linkedinUrl     String?
  crunchbaseUrl   String?
  youtubeUrl      String?
  twitterUrl      String?
  facebookUrl     String?
  instagramUrl    String?
  // Knowledge Panel signals
  wikidataQId     String?                  // e.g. "Q12345"
  kgmid           String?                  // Google KG Machine ID
  kgConfidence    Float?                   // 0.0-1.0 from Google KG API
  // Generated schema
  entityGraph     Json?                    // Full JSON-LD entity graph output
  consistencyScore Float?                  // 0-100
  consistencyReport Json?                  // Last ConsistencyReport
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt
  credentials     BrandCredential[]
  mentions        BrandMention[]
  user            User     @relation(fields: [userId], references: [id])

  @@index([userId])
  @@index([orgId])
}

model BrandCredential {
  id              String   @id @default(cuid())
  brandId         String
  type            String                   // 'award' | 'certification' | 'publication' | 'speaking' | 'membership'
  title           String
  issuer          String
  dateIssued      DateTime
  expiresAt       DateTime?
  url             String?
  description     String?  @db.Text
  verified        Boolean  @default(false)
  createdAt       DateTime @default(now())
  brand           BrandIdentity @relation(fields: [brandId], references: [id], onDelete: Cascade)

  @@index([brandId])
}

model BrandMention {
  id              String   @id @default(cuid())
  brandId         String
  userId          String
  url             String
  urlHash         String                   // SHA-256 of normalised URL for dedup
  title           String
  description     String?  @db.Text
  publishedAt     DateTime?
  source          String                   // Domain of publisher
  apiSource       String                   // 'newsdata' | 'gnews' | 'guardian' | 'gdelt' | 'rss'
  sentiment       String   @default("neutral")  // 'positive' | 'neutral' | 'negative'
  createdAt       DateTime @default(now())
  brand           BrandIdentity @relation(fields: [brandId], references: [id], onDelete: Cascade)

  @@unique([urlHash, brandId])
  @@index([brandId])
  @@index([userId])
  @@index([publishedAt])
}

Generates @id-connected JSON-LD entity graph. Uses sameAs priority ordering.

Key implementation:

- If entityType === 'person': add Person schema referencing the org via 'memberOf'
- articleAuthorRef: minimal { '@id': orgId } for page-level schema references

<task id="4" name="Create lib/brand/consistency-scorer.ts">
Pure TypeScript NAP + handle consistency scorer.

Key implementation:
- `scoreConsistency(identity: BrandIdentityInput): ConsistencyReport`
- For each declared sameAs URL + handle: check if name/handle matches canonical
- Name normalisation: lowercase, strip `\b(inc|llc|ltd|corp|corporation|company|co)\b\.?`, strip `[.,&]`, trim
- NameMatchType: 'exact' (same) | 'variant' (normalized same) | 'mismatch' (different) | 'not-found' (URL missing)
- Platform weights: sameAs platforms (LinkedIn, Wikidata, Wikipedia, Crunchbase) = 3, others = 1
- Score per platform: exact=100, variant=70, mismatch=20, not-found=0
- Weighted average → overallScore
- criticalIssues: platforms with nameMatch === 'mismatch' on sameAs platforms
- Generate recommendations from missing/mismatched platforms
- Note: Does NOT fetch actual platform pages — scores based on declared data only (live checking is expensive)
</task>

<task id="5" name="Create lib/brand/mention-poller.ts">
Multi-API mention aggregator. Queries NewsData.io and GDELT in parallel.

Key implementation:
```typescript
export async function pollMentions(
  brandName: string,
  options?: { apiKeys?: { newsdata?: string; gnews?: string; guardian?: string } }
): Promise<MentionPollResult>
```

- NewsData.io: `https://newsdata.io/api/1/news?apikey=KEY&q="BRAND"&language=en&size=10`
  - Returns `{ results: [{ link, title, description, pubDate, source_id }] }`
  - Skip if `NEWSDATA_API_KEY` env var not set
- GDELT: `https://api.gdeltproject.org/api/v2/doc/doc?query="BRAND"&mode=artlist&maxrecords=10&format=json&sort=datedesc`
  - Returns `{ articles: [{ url, title, seendate, domain }] }`
  - Always available (no auth)
- Call both in parallel with `Promise.allSettled()`
- Map each API's response to `RawMention[]` using inline mapper functions
- Deduplicate using mention-deduplicator.ts
- Return MentionPollResult with totals
</task>

<task id="6" name="Create lib/brand/mention-deduplicator.ts">
URL-based deduplication using canonical URL normalisation.

<task id="7" name="Create lib/brand/wikidata-checker.ts">
Wikidata REST API integration for entity Q-ID lookup and property audit.

Key implementation:
```typescript
export async function checkWikidata(brandName: string, brandUrl?: string): Promise<WikidataCheckResult>
```

- Search: `https://www.wikidata.org/w/api.php?action=wbsearchentities&search=BRAND&language=en&limit=3&format=json`
- Get entity: `https://www.wikidata.org/w/api.php?action=wbgetentities&ids=QXXX&props=labels|claims&format=json`
- Required org props: P31 (instance-of), P856 (official website), P571 (inception date)
- Recommended org props: P159 (HQ location), P112 (founder), P18 (image), P2671 (KGMID)
- Reference count: count `mainsnak.references` arrays across all claims
- Score: 50pts if found + 10pts per required prop + 5pts per recommended prop + up to 10pts for reference count (>3 references)
- Cap at 100
- Return WikidataCheckResult with Q-ID, missing props, score, recommendations
</task>

<task id="8" name="Create lib/brand/kg-confidence-checker.ts">
Google Knowledge Graph Search API wrapper.

```typescript
export async function checkKnowledgeGraph(
  brandName: string,
  apiKey: string
): Promise<KGCheckResult>
```

- URL: `https://kgsearch.googleapis.com/v1/entities:search?query=BRAND&key=KEY&limit=3&indent=True`
- Returns `{ itemListElement: [{ result: { '@id', name, description, '@type': [] }, resultScore }] }`
- Find best match: highest resultScore item whose name fuzzy-matches brandName
- Confidence: resultScore / 1000 clamped to 0-1 (typical scores are 100-1000)
- kgmid: extract from '@id' (format: `kg:/m/XXXXX` → KGMID is `m/XXXXX`)
- Return KGCheckResult
- Skip gracefully if GOOGLE_KG_API_KEY env var not set (return { found: false })
</task>
