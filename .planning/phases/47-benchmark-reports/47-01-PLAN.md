Create service with industry benchmark data and comparison logic:

```typescript
export interface PlatformBenchmarks {
  platform: string;
  metrics: {
    engagementRate: { average: number; good: number; excellent: number };
    followerGrowth: { average: number; good: number; excellent: number };
    postFrequency: { average: number; good: number; excellent: number };
    reachRate: { average: number; good: number; excellent: number };
  };
  byIndustry?: Record<string, Partial<PlatformBenchmarks['metrics']>>;
}

export interface BenchmarkComparison {
  metric: string;
  userValue: number;
  benchmark: { average: number; good: number; excellent: number };
  percentile: number; // 0-100
  rating: 'below' | 'average' | 'good' | 'excellent';
  delta: number; // difference from average
  deltaPercent: number;
}

export interface BenchmarkReport {
  overall: {
    score: number; // 0-100
    rating: 'below' | 'average' | 'good' | 'excellent';
    percentile: number;
  };
  byPlatform: Array<{
    platform: string;
    comparisons: BenchmarkComparison[];
    overallRating: 'below' | 'average' | 'good' | 'excellent';
  }>;
  insights: string[];
  recommendations: string[];
}

export class BenchmarkService {
  getBenchmarks(platform: string, industry?: string): PlatformBenchmarks;
  compareMetrics(userMetrics: UserMetrics, benchmarks: PlatformBenchmarks): BenchmarkComparison[];
  generateReport(userMetrics: UserMetrics[], industry?: string): BenchmarkReport;
}
```

- TikTok: 3-6% engagement (avg), 6-12% (good), 12%+ (excellent)
- YouTube: 2-4% engagement (avg), 4-8% (good), 8%+ (excellent)
- LinkedIn: 2-4% engagement (avg), 4-6% (good), 6%+ (excellent)

Include follower growth benchmarks (monthly %):
