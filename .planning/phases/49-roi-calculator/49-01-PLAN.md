```prisma
model ContentInvestment {
  id          String   @id @default(cuid())
  userId      String

  // Investment details
  type        String   // 'time', 'money'
  category    String   // 'creation', 'equipment', 'software', 'promotion', 'other'
  amount      Decimal  @db.Decimal(12, 2) // hours for time, currency for money
  currency    String?  // Only for money investments
  description String?

  // Optional associations
  platform    String?  // Platform the investment relates to
  postId      String?  // Link to specific content

  // Date
  investedAt  DateTime // When the investment was made

  // Metadata
  metadata    Json?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  // Relations
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId])
  @@index([userId, type])
  @@index([userId, investedAt])
  @@map("content_investments")
}
```

Add relation to User model:

Create ROI calculation service:

```typescript
// Types
export type InvestmentType = 'time' | 'money';
export type InvestmentCategory = 'creation' | 'equipment' | 'software' | 'promotion' | 'other';

export interface ContentInvestment {
  id: string;
  userId: string;
  type: InvestmentType;
  category: InvestmentCategory;
  amount: number;
  currency: string | null;
  description: string | null;
  platform: string | null;
  postId: string | null;
  investedAt: Date;
}

export interface ROIMetrics {
  totalRevenue: number;
  totalMoneyInvested: number;
  totalHoursInvested: number;
  overallROI: number; // percentage
  roiPerHour: number; // revenue per hour
  netProfit: number;
  currency: string;
}

export interface PlatformROI {
  platform: string;
  revenue: number;
  moneyInvested: number;
  hoursInvested: number;
  roi: number;
  roiPerHour: number;
}

export interface ROIReport {
  overall: ROIMetrics;
  byPlatform: PlatformROI[];
  byCategory: Record<InvestmentCategory, number>;
  recentInvestments: ContentInvestment[];
}

// Service class
export class ROIService {
  async getInvestments(userId: string, filters?): Promise<ContentInvestment[]>;
  async createInvestment(userId: string, data): Promise<ContentInvestment>;
  async updateInvestment(id: string, userId: string, data): Promise<ContentInvestment>;
  async deleteInvestment(id: string, userId: string): Promise<void>;
  async calculateROI(userId: string, filters?): Promise<ROIReport>;
}
```

Implementation notes:
- Use RevenueService to get revenue data
- Calculate ROI = (Revenue - Money Investment) / Money Investment * 100
- ROI per hour = Revenue / Hours
- Handle edge cases (zero investment, no revenue)
