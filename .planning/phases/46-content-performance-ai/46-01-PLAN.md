  <name>Task 1: Create content performance analyzer service</name>
  <files>lib/ai/content-performance-analyzer.ts</files>
  <action>
Create service that analyzes post performance patterns:

```typescript
export interface PostPerformance {
  postId: string;
  platform: string;
  content: string;
  hashtags: string[];
  publishedAt: Date;
  metrics: {
    likes: number;
    comments: number;
    shares: number;
    impressions: number;
    engagementRate: number;
  };
}

export interface PerformanceInsight {
  type: 'topic' | 'format' | 'timing' | 'hashtag' | 'length';
  title: string;
  description: string;
  impact: 'high' | 'medium' | 'low';
  recommendation: string;
  evidence: string[];
}

export interface ContentPerformanceAnalysis {
  summary: {
    totalPosts: number;
    avgEngagement: number;
    topPerforming: PostPerformance[];
    lowPerforming: PostPerformance[];
  };
  patterns: {
    bestDays: Array<{ day: string; avgEngagement: number }>;
    bestHours: Array<{ hour: number; avgEngagement: number }>;
    bestLength: { min: number; max: number; avgEngagement: number };
    topHashtags: Array<{ tag: string; avgEngagement: number; count: number }>;
  };
  insights: PerformanceInsight[];
  contentTypes: Array<{
    type: string;
    count: number;
    avgEngagement: number;
    trend: 'up' | 'down' | 'stable';
  }>;
}

export class ContentPerformanceAnalyzer {
  async analyze(posts: PostPerformance[]): Promise<ContentPerformanceAnalysis>;
  async generateAIInsights(analysis: ContentPerformanceAnalysis): Promise<PerformanceInsight[]>;
}
```

**Implementation:**
1. `analyze()`: Pure function that calculates patterns from post data
   - Find top/low performers (top/bottom 10%)
   - Calculate best posting days/hours
   - Identify optimal content length
   - Rank hashtags by engagement
   - Categorize content types (question, statement, story, etc.)

2. `generateAIInsights()`: Uses OpenRouter to generate actionable insights
   - Prompt with top patterns and ask for specific recommendations
   - Return 3-5 high-impact insights
  </action>

<task type="auto">
  <name>Task 2: Create content performance API endpoint</name>
  <files>app/api/content/performance/route.ts</files>
  <action>
Create API endpoint that aggregates and analyzes post performance:

**GET /api/content/performance**
Query params:
- platform: 'all' | specific platform (default: all)
- period: '7d' | '30d' | '90d' (default: 30d)
- includeAI: 'true' | 'false' (default: true)

Response structure matches ContentPerformanceAnalysis interface.

**Implementation:**
1. Auth check using verifyToken pattern
2. Fetch user's PlatformPosts with PlatformMetrics from last N days
3. Transform to PostPerformance[] array
4. Call ContentPerformanceAnalyzer.analyze()
5. If includeAI, call generateAIInsights()
6. Return structured response

**Error handling:**
- No posts: Return empty analysis with helpful message
- AI timeout: Return analysis without AI insights
  </action>
  <verify>API returns performance analysis JSON</verify>
  <done>API returns content performance patterns and insights</done>
</task>
