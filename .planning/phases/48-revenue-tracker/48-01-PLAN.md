```prisma
model RevenueEntry {
  id          String   @id @default(cuid())
  userId      String

  // Revenue details
  source      String   // 'sponsorship', 'affiliate', 'ads', 'tips', 'merchandise', 'other'
  amount      Decimal  @db.Decimal(12, 2)
  currency    String   @default("USD")
  description String?

  // Optional associations
  platform    String?  // Platform that generated the revenue
  postId      String?  // Link to specific content
  brandName   String?  // For sponsorships

  // Dates
  paidAt      DateTime // When payment was received
  periodStart DateTime? // For recurring/period-based income
  periodEnd   DateTime?

  // Metadata
  metadata    Json?    // Source-specific details
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  // Relations
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId])
  @@index([userId, source])
  @@index([userId, paidAt])
  @@map("revenue_entries")
}
```

Add relation to User model:
```prisma
revenueEntries RevenueEntry[]
```

```typescript
// Types
export type RevenueSource = 'sponsorship' | 'affiliate' | 'ads' | 'tips' | 'merchandise' | 'other';

export interface RevenueEntry {
  id: string;
  userId: string;
  source: RevenueSource;
  amount: number;
  currency: string;
  description: string | null;
  platform: string | null;
  postId: string | null;
  brandName: string | null;
  paidAt: Date;
  periodStart: Date | null;
  periodEnd: Date | null;
}

export interface RevenueSummary {
  total: number;
  currency: string;
  bySource: Record<RevenueSource, number>;
  byPlatform: Record<string, number>;
  byMonth: Array<{ month: string; amount: number }>;
  entryCount: number;
  avgPerEntry: number;
  trend: number; // % change from previous period
}

export interface RevenueFilters {
  source?: RevenueSource;
  platform?: string;
  startDate?: Date;
  endDate?: Date;
}

// Service class
export class RevenueService {
  async getSummary(userId: string, filters?: RevenueFilters): Promise<RevenueSummary>;
  async getEntries(userId: string, filters?: RevenueFilters): Promise<RevenueEntry[]>;
  async createEntry(userId: string, data: Omit<RevenueEntry, 'id' | 'userId'>): Promise<RevenueEntry>;
  async updateEntry(id: string, userId: string, data: Partial<RevenueEntry>): Promise<RevenueEntry>;
  async deleteEntry(id: string, userId: string): Promise<void>;
}
```

Implement all methods with Prisma queries.
Calculate trend by comparing current period to previous period of same length.
