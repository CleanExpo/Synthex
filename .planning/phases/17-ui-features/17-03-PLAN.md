### Task 1: Enhance onboarding completion page
**File:** `app/(onboarding)/onboarding/complete/page.tsx`

Read existing file first, then update to:

1. Add celebration animation (confetti already available via lib/celebrations)
2. Show summary of what was configured
3. Add "Take Product Tour" CTA alongside "Go to Dashboard"
4. Set flag to trigger tour on dashboard

```tsx
'use client';

import React, { useEffect } from 'react';
import { useRouter } from 'next/navigation';
import { CheckCircle, ArrowRight, Sparkles, Building, Link2, Brain } from '@/components/icons';
import { Button } from '@/components/ui/button';
import { Card } from '@/components/ui/card';
import { useOnboarding, ProgressIndicator } from '@/components/onboarding';
import { confetti } from '@/lib/celebrations';

const STEPS = [
  { id: 1, name: 'Organization' },
  { id: 2, name: 'Platforms' },
  { id: 3, name: 'Persona' },
  { id: 4, name: 'Complete' },
];

export default function CompletePage() {
  const router = useRouter();
  const { data, markOnboardingComplete } = useOnboarding();

  useEffect(() => {
    // Trigger celebration
    confetti();
    // Mark onboarding complete
    markOnboardingComplete();
    // Set flag to show tour on dashboard
    localStorage.setItem('showTourOnDashboard', 'true');
  }, [markOnboardingComplete]);

  const handleGoToDashboard = (takeTour: boolean) => {
    if (takeTour) {
      // Remove hasSeenTour to trigger tour
      localStorage.removeItem('hasSeenTour');
    }
    router.push('/dashboard');
  };

  const completedItems = [
    { icon: Building, label: 'Organization', value: data.organizationName || 'Set up' },
    { icon: Link2, label: 'Platforms', value: `${data.connectedPlatforms.length} connected` },
    { icon: Brain, label: 'Persona', value: data.personaName || 'Created' },
  ];

  return (
    <div className="space-y-8">
      <ProgressIndicator
        steps={STEPS}
        currentStep={4}
        completedSteps={[1, 2, 3, 4]}
      />

      <div className="text-center space-y-4">
        <div className="w-20 h-20 rounded-full bg-gradient-to-br from-green-500/20 to-emerald-600/10 border border-green-500/20 flex items-center justify-center mx-auto backdrop-blur-sm">
          <CheckCircle className="w-10 h-10 text-green-400" />
        </div>
        <h1 className="text-3xl font-bold text-white">You're All Set!</h1>
        <p className="text-gray-400 max-w-md mx-auto">
          Your SYNTHEX account is ready. Start creating viral content today.
        </p>
      </div>

      {/* Setup Summary */}
      <Card className="max-w-md mx-auto bg-[#0f172a]/80 border-cyan-500/10 p-6">
        <h3 className="text-sm font-medium text-gray-400 mb-4 uppercase tracking-wider">
          Your Setup
        </h3>
        <div className="space-y-3">
          {completedItems.map((item, i) => (
            <div key={i} className="flex items-center justify-between">
              <div className="flex items-center gap-3">
                <div className="w-8 h-8 rounded-lg bg-cyan-500/10 flex items-center justify-center">
                  <item.icon className="w-4 h-4 text-cyan-400" />
                </div>
                <span className="text-gray-300">{item.label}</span>
              </div>
              <span className="text-white font-medium">{item.value}</span>
            </div>
          ))}
        </div>
      </Card>

      {/* CTAs */}
      <div className="flex flex-col sm:flex-row gap-3 justify-center max-w-md mx-auto">
        <Button
          size="lg"
          onClick={() => handleGoToDashboard(true)}
          className="flex-1 bg-gradient-to-r from-cyan-500 to-cyan-600 hover:from-cyan-400 hover:to-cyan-500 text-white shadow-lg shadow-cyan-500/25"
        >
          <Sparkles className="w-4 h-4 mr-2" />
          Take a Quick Tour
        </Button>
        <Button
          size="lg"
          variant="outline"
          onClick={() => handleGoToDashboard(false)}
          className="flex-1 border-cyan-500/30 text-gray-300 hover:bg-cyan-500/10 hover:text-white"
        >
          Skip to Dashboard
          <ArrowRight className="w-4 h-4 ml-2" />
        </Button>
      </div>

      <p className="text-center text-sm text-gray-500">
        You can always take the tour later from the Help menu
      </p>
    </div>
  );
}
```

### Task 3: Update ProductTour to respect showTourOnDashboard flag
**File:** `components/ProductTour.tsx`

Update the useEffect that checks whether to show the tour:

```typescript
useEffect(() => {
  if (typeof window === 'undefined') return;

  const hasSeenTour = localStorage.getItem('hasSeenTour');
  const showTourOnDashboard = localStorage.getItem('showTourOnDashboard');

  // Show tour if:
  // 1. User just completed onboarding and wants tour
  // 2. User hasn't seen tour and has completed onboarding
  if (showTourOnDashboard === 'true') {
    localStorage.removeItem('showTourOnDashboard');
    setTimeout(() => setIsActive(true), 500);
  } else if (!hasSeenTour) {
    const onboardingComplete = localStorage.getItem('onboardingComplete');
    if (onboardingComplete) {
      setTimeout(() => setIsActive(true), 1000);
    }
  }
}, []);
```
