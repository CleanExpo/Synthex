**Architecture rules:**
- Auth: Supabase ONLY
- No mock data — every email sends via real Resend SDK
- Australian English in email copy (colour, recognise, organise)
- Resend SDK already installed — see existing weekly digest pattern in lib/email/

**Email pattern (from Phase 6 weekly digest):**
- Resend client: `import { Resend } from 'resend';` → `new Resend(process.env.RESEND_API_KEY)`
- From address: `process.env.EMAIL_FROM` (e.g. "Synthex <noreply@synthex.social>")
- All email calls: fire-and-forget (don't await — billing webhook must respond in <3s)
- Pattern: `resend.emails.send({...}).catch(err => console.error('Email send failed:', err))`

    ```typescript
    import { Resend } from 'resend';
    const resend = new Resend(process.env.RESEND_API_KEY);

    export function sendPaymentReceiptEmail(params: {
      email: string; name?: string; amount: number; currency: string;
      plan: string; billingPortalUrl: string;
    }): void {
      resend.emails.send({
        from: process.env.EMAIL_FROM ?? 'Synthex <noreply@synthex.social>',
        to: params.email,
        subject: `Payment received — ${params.plan} plan`,
        html: `<p>Hi ${params.name ?? 'there'},</p>
               <p>Thank you for your payment of ${params.currency.toUpperCase()} ${(params.amount / 100).toFixed(2)}.</p>
               <p>Your ${params.plan} subscription is now active.</p>
               <p><a href="${params.billingPortalUrl}">Manage your subscription</a></p>`,
      }).catch(err => console.error('[billing-email] receipt send failed:', err));
    }

    export function sendPaymentFailedEmail(params: {
      email: string; name?: string; amount: number; currency: string;
      billingPortalUrl: string;
    }): void {
      resend.emails.send({
        from: process.env.EMAIL_FROM ?? 'Synthex <noreply@synthex.social>',
        to: params.email,
        subject: 'Payment failed — action required',
        html: `<p>Hi ${params.name ?? 'there'},</p>
               <p>We were unable to process your payment of ${params.currency.toUpperCase()} ${(params.amount / 100).toFixed(2)}.</p>
               <p>Please <a href="${params.billingPortalUrl}">update your payment method</a> to keep your subscription active.</p>`,
      }).catch(err => console.error('[billing-email] failed-payment send failed:', err));
    }

    export function sendSubscriptionCancelledEmail(params: {
      email: string; name?: string; plan: string; endDate: string;
    }): void {
      resend.emails.send({
        from: process.env.EMAIL_FROM ?? 'Synthex <noreply@synthex.social>',
        to: params.email,
        subject: 'Your Synthex subscription has been cancelled',
        html: `<p>Hi ${params.name ?? 'there'},</p>
               <p>Your ${params.plan} subscription has been cancelled.</p>
               <p>You'll have access to your paid features until ${params.endDate}.</p>
               <p>We'd love to have you back — <a href="https://synthex.social/pricing">view plans</a>.</p>`,
      }).catch(err => console.error('[billing-email] cancellation send failed:', err));
    }
    ```
