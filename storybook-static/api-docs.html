Getting Started with SYNTHEX API

🔐 Authentication

Use JWT Bearer tokens in the Authorization header:

Authorization: Bearer <token>

🚀 Base URL

Production API endpoint:

https://synthex.social/api/v1

⚡ Rate Limits

Requests per hour by plan:

Free: 100 | Pro: 1000 | Enterprise: ∞

📦 SDKs Available

Official libraries for:

JavaScript | Python | PHP | Ruby

Quick Start Examples

# Authentication
curl -X POST https://synthex.social/api/v1/auth/login \
  -H "Content-Type: application/json" \
  -d '{"email":"user@example.com","password":"your-password"}'

# Create a post
curl -X POST https://synthex.social/api/v1/posts \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "title": "My First Post",
    "content": "Hello from SYNTHEX API!",
    "platforms": ["twitter", "facebook"]
  }'

# Get analytics
curl -X GET "https://synthex.social/api/v1/analytics/overview?range=week" \
  -H "Authorization: Bearer YOUR_TOKEN"
// Install: npm install @synthex/api-sdk

import { SynthexAPI } from '@synthex/api-sdk';

const api = new SynthexAPI({
  apiKey: 'YOUR_API_KEY'
});

// Create a post
const post = await api.posts.create({
  title: 'My First Post',
  content: 'Hello from SYNTHEX API!',
  platforms: ['twitter', 'facebook'],
  scheduledAt: new Date('2024-01-01 10:00:00')
});

// Get analytics
const analytics = await api.analytics.getOverview({
  range: 'week'
});

// Listen to webhooks
api.webhooks.on('post.published', (data) => {
  console.log('Post published:', data);
});
# Install: pip install synthex-api

from synthex import SynthexAPI
from datetime import datetime

api = SynthexAPI(api_key='YOUR_API_KEY')

# Create a post
post = api.posts.create(
    title='My First Post',
    content='Hello from SYNTHEX API!',
    platforms=['twitter', 'facebook'],
    scheduled_at=datetime(2024, 1, 1, 10, 0)
)

# Get analytics
analytics = api.analytics.get_overview(range='week')

# Bulk operations
posts = [
    {'title': f'Post {i}', 'content': f'Content {i}', 'platforms': ['twitter']}
    for i in range(10)
]
results = api.posts.create_bulk(posts)
